struct node
{
	int data;
	struct node *next;
	
}*num,*head,*tail,*temp;

int traversal()
{
	int counter=0;
	temp = head;
	printf("The Elements in the List is : ");
	while(temp != NULL)
	{
		printf(" %d ->",temp->data);
		temp = temp->next;
		counter += 1;
	}
	printf("\nThe Number of Nodes in the Linked List is : %d",counter);
	return 0;
}

int creation()
{
	int n,i,value;
	printf("Enter Number of Elements :  ");
	scanf("%d",&n);
	printf("Enter the Elements into the List : ");
	for(i=1;i<=n;i++)
	{
		num  = (struct node*) malloc(sizeof(struct node));
		scanf("%d",&value);
		num->data = value;
		num->next = NULL;
		if(head == NULL)
		{
			head = num;
			tail = num;
		}
		else
		{
			tail->next = num;
			tail = num;
		}
	}
	printf("\n**The Elements Has been Created Succesfully**");
	return 0;
}
int insertbeg()
{
	printf("Enter the Element to be Inserted at the Beggining : ");
	num = (struct node*)malloc(sizeof(struct node));
	scanf("%d",&num->data);
	if(head == NULL)
	{
		head = num;
		tail = num;
	}
	else
	{
		num->next = head;
		head = num;
	}
	printf("**The Element has been Inserted at the Beginning Sucessfully**");
}
int insertend()
{
	printf("Enter the Element to be Inserted at the End : ");
	num = (struct node*)malloc(sizeof(struct node));
	scanf("%d",&num->data);
	num->next = NULL;
	if(tail == NULL)
	{
		tail = num;
		tail->next = NULL;
	}
	else
	{
		tail->next  = num;
	}
	printf("**The Element has been Inserted at the End Sucessfully**");

}
int choice()
{
	int c;
	printf("\nEnter Your Choice \n\t(1).Creation \n\t(2).Traversal \n\t(3).Insertion at the Begining \n\t(4).Insertion at the Ending \n\t(5).Exit : ");
	scanf("%d",&c);
	switch (c)
	{
		case 1: 
		    creation();

		case 2:

		case 5:

		default:

	}
			
}
